    // ========== CANVAS SETUP ==========
    function resizeCanvas() {
        const rect = diagramEl.getBoundingClientRect();
        const dpr = window.devicePixelRatio || 1;
        canvas.width = rect.width * dpr;
        canvas.height = rect.height * dpr;
        canvas.style.width = rect.width + 'px';
        canvas.style.height = rect.height + 'px';
        ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
    }

    // Lấy tâm của node so với canvas
    function getNodeCenter(id) {
        const node = document.getElementById(id);
        if (!node) return null;
        const cRect = canvas.getBoundingClientRect();
        const nRect = node.getBoundingClientRect();
        return {
            x: nRect.left + nRect.width / 2 - cRect.left,
            y: nRect.top + nRect.height / 2 - cRect.top,
            w: nRect.width,
            h: nRect.height
        };
    }

    // Lấy cạnh của node (left/right/top/bottom center)
    function getNodeEdge(id, side) {
        const c = getNodeCenter(id);
        if (!c) return null;
        switch(side) {
            case 'right': return { x: c.x + c.w/2, y: c.y };
            case 'left': return { x: c.x - c.w/2, y: c.y };
            case 'top': return { x: c.x, y: c.y - c.h/2 };
            case 'bottom': return { x: c.x, y: c.y + c.h/2 };
            default: return { x: c.x, y: c.y };
        }
    }

    // Xác định hướng kết nối giữa 2 node
    function getEdgePair(fromId, toId) {
        const fc = getNodeCenter(fromId);
        const tc = getNodeCenter(toId);
        if (!fc || !tc) return null;
        const dx = tc.x - fc.x;
        const dy = tc.y - fc.y;

        // Nếu gần như ngang ( bên trái)
        if (Math.abs(dx) > Math.abs(dy) * 0.5) {
            return {
                from: getNodeEdge(fromId, dx > 0 ? 'right' : 'left'),
                to: getNodeEdge(toId, dx > 0 ? 'left' : 'right')
            };
        }
        // Nếu gần như dọc
        return {
            from: getNodeEdge(fromId, dy > 0 ? 'bottom' : 'top'),
            to: getNodeEdge(toId, dy > 0 ? 'top' : 'bottom')
        };
    }

    // Tạo bezier curve points
    function getBezierPoints(p0, p3) {
        const dx = p3.x - p0.x;
        const dy = p3.y - p0.y;
        const cp1 = { x: p0.x + dx * 0.45, y: p0.y };
        const cp2 = { x: p3.x - dx * 0.45, y: p3.y };
        return { p0, cp1, cp2, p3 };
    }

    // Interpolate trên bezier
    function bezierPoint(t, p0, cp1, cp2, p3) {
        const u = 1 - t;
        return {
            x: u*u*u*p0.x + 3*u*u*t*cp1.x + 3*u*t*t*cp2.x + t*t*t*p3.x,
            y: u*u*u*p0.y + 3*u*u*t*cp1.y + 3*u*t*t*cp2.y + t*t*t*p3.y
        };
    }

    // ========== PARTICLE SYSTEM ==========
    function spawnParticles() {
        particles = [];
        const activeConns = connections.filter(c => c.scenarios.includes(currentScenario));

        activeConns.forEach((conn, idx) => {
            const edgePair = getEdgePair(conn.from, conn.to);
            if (!edgePair) return;
            const bez = getBezierPoints(edgePair.from, edgePair.to);

            // Request particles (xanh lá)
            for (let i = 0; i < 3; i++) {
                particles.push({
                    bez: bez,
                    progress: -i * 0.25 - Math.random() * 0.1,
                    speed: 0.004 + Math.random() * 0.002,
                    color: '#00e5a0',
                    type: 'request',
                    size: 4,
                    connIdx: idx
                });
            }
            // Response particles (vàng) — đi ngược lại, delay hơn
            for (let i = 0; i < 2; i++) {
                particles.push({
                    bez: { p0: bez.p3, cp1: bez.cp2, cp2: bez.cp1, p3: bez.p0 },
                    progress: -i * 0.3 - 0.4 - Math.random() * 0.1,
                    speed: 0.0035 + Math.random() * 0.002,
                    color: '#f5a623',
                    type: 'response',
                    size: 3.5,
                    connIdx: idx
                });
            }
        });
    }

    function updateAndDrawParticles() {
        const w = parseFloat(canvas.style.width);
        const h = parseFloat(canvas.style.height);
        ctx.clearRect(0, 0, w, h);

        // Vẽ đường kết nối mờ
        const activeConns = connections.filter(c => c.scenarios.includes(currentScenario));
        activeConns.forEach(conn => {
            const edgePair = getEdgePair(conn.from, conn.to);
            if (!edgePair) return;
            const bez = getBezierPoints(edgePair.from, edgePair.to);

            ctx.beginPath();
            ctx.moveTo(bez.p0.x, bez.p0.y);
            ctx.bezierCurveTo(bez.cp1.x, bez.cp1.y, bez.cp2.x, bez.cp2.y, bez.p3.x, bez.p3.y);
            ctx.strokeStyle = 'rgba(0,229,160,0.08)';
            ctx.lineWidth = 2;
            ctx.stroke();

            // Vẽ mũi tên ở giữa path
            const mid = bezierPoint(0.5, bez.p0, bez.cp1, bez.cp2, bez.p3);
            const ahead = bezierPoint(0.52, bez.p0, bez.cp1, bez.cp2, bez.p3);
            const angle = Math.atan2(ahead.y - mid.y, ahead.x - mid.x);
            ctx.save();
            ctx.translate(mid.x, mid.y);
            ctx.rotate(angle);
            ctx.beginPath();
            ctx.moveTo(6, 0);
            ctx.lineTo(-4, -4);
            ctx.lineTo(-4, 4);
            ctx.closePath();
            ctx.fillStyle = 'rgba(0,229,160,0.15)';
            ctx.fill();
            ctx.restore();
        });

        // Cập nhật và vẽ particles
        particles.forEach(p => {
            p.progress += p.speed;
            if (p.progress > 1.3) {
                p.progress = -Math.random() * 0.4;
            }
            if (p.progress < 0 || p.progress > 1) return;

            const pos = bezierPoint(p.progress, p.bez.p0, p.bez.cp1, p.bez.cp2, p.bez.p3);
            const alpha = p.progress < 0.1 ? p.progress / 0.1 : (p.progress > 0.9 ? (1 - p.progress) / 0.1 : 1);

            // Vẽ trail
            for (let t = 1; t <= 5; t++) {
                const tp = Math.max(0, p.progress - t * 0.015);
                const tpos = bezierPoint(tp, p.bez.p0, p.bez.cp1, p.bez.cp2, p.bez.p3);
                const ta = alpha * (1 - t / 6) * 0.3;
                ctx.beginPath();
                ctx.arc(tpos.x, tpos.y, p.size * (1 - t * 0.12), 0, Math.PI * 2);
                ctx.fillStyle = p.color.replace(')', `,${ta})`).replace('rgb', 'rgba').replace('#', '');
                // Chuyển hex sang rgba
                const r = parseInt(p.color.slice(1,3),16);
                const g = parseInt(p.color.slice(3,5),16);
                const b = parseInt(p.color.slice(5,7),16);
                ctx.fillStyle = `rgba(${r},${g},${b},${ta})`;
                ctx.fill();
            }

            // Vẽ particle chính với glow
            const r = parseInt(p.color.slice(1,3),16);
            const g = parseInt(p.color.slice(3,5),16);
            const b = parseInt(p.color.slice(5,7),16);

            // Glow
            const grad = ctx.createRadialGradient(pos.x, pos.y, 0, pos.x, pos.y, p.size * 3);
            grad.addColorStop(0, `rgba(${r},${g},${b},${alpha * 0.5})`);
            grad.addColorStop(1, `rgba(${r},${g},${b},0)`);
            ctx.beginPath();
            ctx.arc(pos.x, pos.y, p.size * 3, 0, Math.PI * 2);
            ctx.fillStyle = grad;
            ctx.fill();

            // Core
            ctx.beginPath();
            ctx.arc(pos.x, pos.y, p.size, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(${r},${g},${b},${alpha})`;
            ctx.fill();

            // White center
            ctx.beginPath();
            ctx.arc(pos.x, pos.y, p.size * 0.4, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(255,255,255,${alpha * 0.8})`;
            ctx.fill();
        });
    }

    // ========== SCENARIO CONTROL ==========
    function setScenario(scenario) {
        currentScenario = scenario;

        // Cập nhật buttons
        document.querySelectorAll('.scenario-btn').forEach(btn => {
            btn.classList.toggle('active', btn.dataset.scenario === scenario);
        });

        // Cập nhật node states
        const allNodeIds = ['tv1','tv2','tv3','admin-pc','vod-source','switch','proxy','guest-ui','admin-portal','backend','db','cache'];
        const activeConns = connections.filter(c => c.scenarios.includes(scenario));
        const activeNodeIds = new Set();
        activeConns.forEach(c => { activeNodeIds.add(c.from); activeNodeIds.add(c.to); });

        allNodeIds.forEach(id => {
            const el = document.getElementById(id);
            if (!el) return;
            el.classList.remove('active', 'dimmed');
            if (scenario === 'all') {
                el.classList.add('active');
            } else if (activeNodeIds.has(id)) {
                el.classList.add('active');
            } else {
                el.classList.add('dimmed');
            }
        });

        // Spawn particles mới
        spawnParticles();

        // Cập nhật step bar
        renderSteps(scenario);

        // Reset step highlight timer
        resetStepHighlight();
    }

    function renderSteps(scenario) {
        const bar = document.getElementById('step-bar');
        const steps = scenarioSteps[scenario] || [];
        bar.innerHTML = steps.map((s, i) => `
            <div class="step-item" id="step-${i}">
                <div class="step-num">${i + 1}</div>
                <div class="step-text">${s.text}</div>
            </div>
        `).join('');
    }

    function resetStepHighlight() {
        if (stepTimer) clearInterval(stepTimer);
        stepHighlightIndex = 0;
        const steps = scenarioSteps[currentScenario] || [];
        if (steps.length === 0) return;

        // Highlight bước đầu tiên
        highlightStep(0);

        stepTimer = setInterval(() => {
            stepHighlightIndex = (stepHighlightIndex + 1) % steps.length;
            highlightStep(stepHighlightIndex);
        }, 2500);
    }

    function highlightStep(idx) {
        const steps = scenarioSteps[currentScenario] || [];
        steps.forEach((_, i) => {
            const el = document.getElementById(`step-${i}`);
            if (el) el.classList.toggle('highlight', i === idx);
        });

        // Highlight nodes tương ứng
        const allNodeIds = ['tv1','tv2','tv3','admin-pc','vod-source','switch','proxy','guest-ui','admin-portal','backend','db','cache'];
        const stepNodes = new Set(steps[idx]?.nodes || []);

        allNodeIds.forEach(id => {
            const el = document.getElementById(id);
            if (!el || el.classList.contains('dimmed')) return;
            if (stepNodes.has(id)) {
                el.style.boxShadow = '0 0 28px rgba(0,229,160,0.35), inset 0 0 16px rgba(0,229,160,0.08)';
                el.style.borderColor = '#00e5a0';
            } else if (currentScenario !== 'all') {
                el.style.boxShadow = '';
                el.style.borderColor = '';
            }
        });
    }

    function closeInfo() {
        document.getElementById('info-panel').classList.remove('open');
        document.getElementById('overlay').style.display = 'none';
    }

    // ========== ANIMATION LOOP ==========
    function animate() {
        updateAndDrawParticles();
        animFrameId = requestAnimationFrame(animate);
    }

    // ========== INIT ==========
    function init() {
        resizeCanvas();
        setScenario('guest');
        animate();

        // Resize handler
        let resizeTimeout;
        window.addEventListener('resize', () => {
            clearTimeout(resizeTimeout);
            resizeTimeout = setTimeout(() => {
                resizeCanvas();
                spawnParticles();
            }, 150);
        });

        // Keyboard: Escape đóng panel
        document.addEventListener('keydown', e => {
            if (e.key === 'Escape') closeInfo();
        });
    }

    // Đợi DOM và fonts load
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', () => setTimeout(init, 100));
    } else {
        setTimeout(init, 100);
    }
